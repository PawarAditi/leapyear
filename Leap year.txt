#include <iostream>

using namespace std;

int main()
{
    cout<<"Leap year\n";
    int yr;
    cout<<"Enter year : \n";
    cin>>yr;
    if(yr%400 == 0){
        cout<<"Year is leap year \n";
    }
    else if(yr % 100 == 0){
        cout<<"Year is not leap year ";
    }
    else if(yr % 4 == 0){
        cout<<"Year is leap year ";
    }
    else{
        cout<<"Year is not leap year ";
    }
    return 0;
}
